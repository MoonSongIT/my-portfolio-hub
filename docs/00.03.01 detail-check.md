export default function Dashboard() {
  const { holdings, cash, addHolding } = usePortfolioStore()

  const handleTestAdd = () => {
    addHolding({
      ticker: 'TEST',
      name: '테스트 종목',

      current_price: 11000,

      <p>현금: {cash.toLocaleString()}원</p>

    </div>
  )
}
