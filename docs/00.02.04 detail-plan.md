### Step 1: 로컬 개발용 Express 프록시 서버 구축

**목표**: 로컬 개발 환경에서 Claude API 키를 서버사이드에서 안전하게 보호하는 Express 서버 구축

**생성 파일**: `server/index.js`

```javascript
// server/index.js
import express from 'express'
import cors from 'cors'
import fetch from 'node-fetch'
import dotenv from 'dotenv'

dotenv.config()

const app = express()
app.use(cors({ origin: 'http://localhost:5173' }))
app.use(express.json())

// Claude API 프록시 엔드포인트
app.post('/api/claude', async (req, res) => {
  const { systemPrompt, messages, maxTokens = 4096 } = req.body

  if (!process.env.ANTHROPIC_API_KEY) {
    return res.status(500).json({ error: 'API 키 미설정' })
  }

  try {
    const response = await fetch('https://api.anthropic.com/v1/messages', {
      method: 'POST',
      headers: {
        'Content-Type': 'application/json',
        'x-api-key': process.env.ANTHROPIC_API_KEY,
        'anthropic-version': '2023-06-01',
      },
      body: JSON.stringify({
        model: 'claude-sonnet-4-6',
        max_tokens: maxTokens,
        system: systemPrompt,
        messages,
      }),
    })

    if (!response.ok) {
      const errText = await response.text()
      return res.status(response.status).json({ error: errText })
    }

    const data = await response.json()
    res.json(data)
  } catch (err) {
    res.status(500).json({ error: err.message })
  }
})

app.listen(3001, () => console.log('🤖 Claude 프록시 서버: http://localhost:3001'))
```

### Step 2: Vercel Serverless Function 구현

**목표**: 프로덕션 배포 시 Anthropic API 키를 서버사이드에서 보호하는 Vercel Function 구현

**생성 파일**: `api/claude.js`

```javascript
// api/claude.js — Vercel Serverless Function
export default async function handler(req, res) {
  // CORS 헤더
  res.setHeader('Access-Control-Allow-Origin', '*')
  res.setHeader('Access-Control-Allow-Methods', 'POST, OPTIONS')
  res.setHeader('Access-Control-Allow-Headers', 'Content-Type')

  if (req.method === 'OPTIONS') return res.status(200).end()
  if (req.method !== 'POST') return res.status(405).json({ error: 'Method Not Allowed' })

  const { systemPrompt, messages, maxTokens = 4096 } = req.body

  if (!systemPrompt || !messages?.length) {
    return res.status(400).json({ error: '필수 파라미터 누락 (systemPrompt, messages)' })
  }

  try {
    const response = await fetch('https://api.anthropic.com/v1/messages', {
      method: 'POST',
      headers: {
        'Content-Type': 'application/json',
        'x-api-key': process.env.ANTHROPIC_API_KEY,
        'anthropic-version': '2023-06-01',
      },
      body: JSON.stringify({
        model: 'claude-sonnet-4-6',
        max_tokens: maxTokens,
        system: systemPrompt,
        messages,
      }),
    })

    if (!response.ok) {
      const err = await response.text()
      return res.status(response.status).json({ error: err })
    }

    const data = await response.json()
    res.status(200).json(data)
  } catch (err) {
    res.status(500).json({ error: '서버 오류: ' + err.message })
  }
}
```

  { agent: 'research',   keywords: ['분석', '어때', '살까', '종목', '주가', '전망', '리서치', 'analysis'] },
  { agent: 'portfolio',  keywords: ['포트폴리오', '내 종목', '수익률', '현황', '보유', '리밸런싱', '비중'] },
  { agent: 'alert',      keywords: ['관심종목', '오늘 시장', '알림', '체크', '브리핑', '이상', '급등', '급락'] },
  { agent: 'report',     keywords: ['리포트', '성과', '이번달', '결산', '실적', '기간', '수익'] },

// 사용자 메시지 → 에이전트 타입 라우팅
export function routeToAgent(userMessage) {
  const msg = userMessage.toLowerCase()
  for (const rule of ROUTING_RULES) {
    if (rule.keywords.some(kw => msg.includes(kw))) return rule.agent
  }
  return 'portfolio' // 기본값
}

// 포트폴리오 컨텍스트 빌더
export function buildPortfolioContext(holdings, exchangeRate) {
  if (!holdings?.length) return '[보유 종목 없음]'

  const lines = holdings.map(h => {
    const currPrice = h.currentPrice ?? h.avgPrice
    const profit = (currPrice - h.avgPrice) * h.quantity
    const profitRate = ((currPrice - h.avgPrice) / h.avgPrice * 100).toFixed(2)
    return `  - ${h.name}(${h.ticker}): ${h.quantity}주, 평균${h.avgPrice.toLocaleString()}, 현재${currPrice.toLocaleString()}, 수익률${profitRate}%`
  }).join('\n')

  return `
[보유 포트폴리오 현황]
환율: 1 USD = ${exchangeRate} KRW
종목 수: ${holdings.length}개

${lines}
`.trim()
}

// 관심종목 컨텍스트 빌더
export function buildAlertContext(watchlist, quotesMap) {
  if (!watchlist?.length) return '[관심종목 없음]'

  const lines = watchlist.map(w => {
    const q = quotesMap?.[w.ticker]
    const change = q?.changePercent?.toFixed(2) ?? '?'
    const level = Math.abs(parseFloat(change)) >= 10 ? '🔴' :
                  Math.abs(parseFloat(change)) >= 5  ? '🟡' : '🟢'
    return `  ${level} ${w.name}(${w.ticker}): ${change}%`
  }).join('\n')

  return `
[관심종목 시세 현황]
기준 시각: ${new Date().toLocaleString('ko-KR')}

${lines}
`.trim()
}

```javascript
// src/api/claudeApi.js
import axios from 'axios'
import { routeToAgent, ORCHESTRATOR_PROMPT, AGENT_LABELS } from '../agents/orchestrator'
import { RESEARCH_SYSTEM_PROMPT, buildResearchContext } from '../agents/researchAgent'
import { PORTFOLIO_SYSTEM_PROMPT, buildPortfolioContext } from '../agents/portfolioAgent'
import { ALERT_SYSTEM_PROMPT, buildAlertContext } from '../agents/alertAgent'
import { REPORT_SYSTEM_PROMPT, buildReportContext } from '../agents/reportAgent'

const claudeApi = axios.create({
  baseURL: import.meta.env.VITE_API_BASE_URL || '/api',
  timeout: 60_000,
})

// 에이전트 타입 → 시스템 프롬프트 매핑
const AGENT_PROMPTS = {
  research:  RESEARCH_SYSTEM_PROMPT,
  portfolio: PORTFOLIO_SYSTEM_PROMPT,
  alert:     ALERT_SYSTEM_PROMPT,
  report:    REPORT_SYSTEM_PROMPT,
}

// 에이전트 타입 → 컨텍스트 빌더 매핑
const CONTEXT_BUILDERS = {
  research:  (ctx) => buildResearchContext(ctx.stockData),
  portfolio: (ctx) => buildPortfolioContext(ctx.holdings, ctx.exchangeRate),
  alert:     (ctx) => buildAlertContext(ctx.watchlist, ctx.quotesMap),
  report:    (ctx) => buildReportContext(ctx.holdings, ctx.period, ctx.exchangeRate),
}

/**
 * AI 에이전트에 메시지 전송
 * @param {string} userMessage  - 사용자 입력
 * @param {object} context      - 포트폴리오/종목 등 컨텍스트 데이터
 * @param {string} [forceAgent] - 강제 에이전트 지정 (선택)
 * @returns {Promise<{text: string, agentType: string}>}
 */
export async function sendToAgent(userMessage, context = {}, forceAgent = null) {
  const agentType = forceAgent || routeToAgent(userMessage)
  const systemPrompt = AGENT_PROMPTS[agentType] || ORCHESTRATOR_PROMPT
  const contextText  = CONTEXT_BUILDERS[agentType]?.(context) || ''

  const fullUserMessage = contextText
    ? `${contextText}\n\n사용자 질문: ${userMessage}`
    : userMessage

  const { data } = await claudeApi.post('/claude', {
    systemPrompt,
    messages: [{ role: 'user', content: fullUserMessage }],
    maxTokens: 4096,
  })

  const text = data?.content?.[0]?.text || '응답을 받지 못했습니다.'
  return { text, agentType, agentInfo: AGENT_LABELS[agentType] }
}

/**
 * 다중 턴 대화 전송 (채팅 히스토리 포함)
 */
export async function sendWithHistory(messages, systemPrompt) {
  const { data } = await claudeApi.post('/claude', {
    systemPrompt,
    messages,
    maxTokens: 4096,
  })
  return data?.content?.[0]?.text || '응답을 받지 못했습니다.'
}

export const useChatStore = create(
  persist(
    (set, get) => ({
      // 대화 메시지 목록
      messages: [],
      // 현재 처리 중 여부
      isLoading: false,
      // 마지막으로 사용된 에이전트
      lastAgentType: null,
      // 에러 메시지
      error: null,

      // 사용자 메시지 추가

        messages: [...state.messages, {
          id: Date.now(),
          role: 'user',
          text,
          timestamp: new Date().toISOString(),
        }],
        error: null,
      })),

      // AI 응답 메시지 추가
      addAIMessage: (text, agentType, agentInfo) => set(state => ({
        messages: [...state.messages, {
          id: Date.now(),
          role: 'assistant',
          text,
          agentType,
          agentInfo,
          timestamp: new Date().toISOString(),
        }],
        lastAgentType: agentType,
      })),

      // 로딩 상태 토글
      setLoading: (isLoading) => set({ isLoading }),

      // 에러 설정
      setError: (error) => set({ error, isLoading: false }),

      // 대화 초기화
      clearHistory: () => set({ messages: [], error: null, lastAgentType: null }),
    }),
    {
      name: 'chat-history',
      partialize: (state) => ({ messages: state.messages.slice(-50) }), // 최근 50개만 유지

// 에러 메시지 사용자 친화 변환
function parseClaudeError(err) {
  if (!err.response) return '네트워크 연결을 확인해주세요.'
  const status = err.response.status
  if (status === 429) return 'AI 요청 한도에 도달했습니다. 잠시 후 다시 시도해주세요.'
  if (status === 401) return 'API 키 오류입니다. 관리자에게 문의하세요.'
  if (status === 500) return 'AI 서버 오류가 발생했습니다. 잠시 후 다시 시도해주세요.'
  if (err.code === 'ECONNABORTED') return '응답 시간이 초과되었습니다 (60초). 질문을 간단하게 해보세요.'
  return err.response?.data?.error || '알 수 없는 오류가 발생했습니다.'
}

// sendToAgent 내 retry 로직 (최대 2회)
export async function sendToAgent(userMessage, context = {}, forceAgent = null, retryCount = 0) {
  try {
    // ... 기존 로직
  } catch (err) {
    if (retryCount < 2 && err.response?.status >= 500) {
      await new Promise(r => setTimeout(r, 1000 * (retryCount + 1)))
      return sendToAgent(userMessage, context, forceAgent, retryCount + 1)
    }
    throw new Error(parseClaudeError(err))
  }
}
```
