### 1.3 CandlestickChart 컴포넌트 구현

**경로**: `src/components/charts/CandlestickChart.jsx`

```javascript
import { useEffect, useRef } from 'react';
import { createChart } from 'lightweight-charts';

export default function CandlestickChart({ data = [], ticker = '', timeframe = '1D' }) {
  const containerRef = useRef(null);
  const chartRef = useRef(null);

  useEffect(() => {
    if (!containerRef.current || data.length === 0) return;

    // 차트 생성
    const chart = createChart(containerRef.current, {
      layout: {
        textColor: '#333',
        background: { type: 'solid', color: '#fff' },
      },
      timeScale: {
        timeVisible: true,
        secondsVisible: false,
      },
      width: containerRef.current.clientWidth,
      height: 500,
    });

    // 캔들스틱 시리즈
    const candleSeriesOptions = {
      upColor: '#26a69a',      // 양봉 (상승) 색상
      downColor: '#ef5350',    // 음봉 (하강) 색상
      borderDownColor: '#ef5350',
      borderUpColor: '#26a69a',
      wickDownColor: '#ef5350',
      wickUpColor: '#26a69a',
    };
    const candleSeries = chart.addCandlestickSeries(candleSeriesOptions);
    candleSeries.setData(data);

    // 기술 지표: 20일, 60일, 120일 이동평균선
    const sma20 = calculateSMA(data, 20);
    const sma60 = calculateSMA(data, 60);
    const sma120 = calculateSMA(data, 120);

    const lineSMA20 = chart.addLineSeries({
      color: '#FF9800',
      lineWidth: 2,
      title: 'SMA 20',
    });
    lineSMA20.setData(sma20);

    const lineSMA60 = chart.addLineSeries({
      color: '#2196F3',
      lineWidth: 2,
      title: 'SMA 60',
    });
    lineSMA60.setData(sma60);

    const lineSMA120 = chart.addLineSeries({
      color: '#9C27B0',
      lineWidth: 2,
      title: 'SMA 120',
    });
    lineSMA120.setData(sma120);

    // 거래량 히스토그램 (하단)
    const volumeSeries = chart.addHistogramSeries({
      color: '#26a69a',
      lineWidth: 1,
      priceFormat: {
        type: 'volume',
      },
    });

    const volumeData = data.map(candle => ({
      time: candle.time,
      value: candle.volume,
      color: candle.close >= candle.open ? '#26a69a' : '#ef5350',
    }));
    volumeSeries.setData(volumeData);

    // 시간축 자동 스케일
    chart.timeScale().fitContent();

    // 창 리사이징 시 차트 재조정
    const handleResize = () => {
      if (containerRef.current) {
        chart.applyOptions({
          width: containerRef.current.clientWidth,
        });
      }
    };
    window.addEventListener('resize', handleResize);

    chartRef.current = chart;

    return () => {
      window.removeEventListener('resize', handleResize);
      chart.remove();
    };
  }, [data]);

  return (
    <div className="w-full bg-white rounded-lg border border-gray-200">
      <div className="px-4 py-2 border-b border-gray-200">
        <h3 className="text-lg font-semibold">{ticker} {timeframe} 차트</h3>
      </div>
      <div ref={containerRef} style={{ width: '100%' }} />
    </div>
  );
}

// 이동평균선 계산 헬퍼
function calculateSMA(data, period) {
  return data
    .map((candle, index) => {
      if (index < period - 1) return null;
      const sum = data.slice(index - period + 1, index + 1).reduce(
        (acc, c) => acc + c.close,
        0
      );
      return {
        time: candle.time,
        value: sum / period,
      };
    })
    .filter(Boolean);
}
```

### 2.2 Database 스키마 정의

**경로**: `src/utils/db.js`

```javascript
import Dexie from 'dexie';

export const db = new Dexie('PortfolioHub');

db.version(1).stores({
  // 키: '++id'는 자동증분, 'ticker'/'date'는 인덱스
  transactions: '++id, ticker, type, date, accountId',
  priceHistory: '++id, ticker, date',
  reports: '++id, type, dateRange, createdAt',
});

// TypeScript 타입 힌트 (선택)
export async function addTransaction(transaction) {
  return db.transactions.add(transaction);
}

export async function getTransactionsByTicker(ticker) {
  return db.transactions.where('ticker').equals(ticker).toArray();
}

export async function getTransactionsByDate(startDate, endDate) {
  return db.transactions
    .where('date')
    .between(startDate, endDate)
    .toArray();
}

export async function getPriceHistory(ticker, days = 365) {
  const startDate = new Date();
  startDate.setDate(startDate.getDate() - days);

  return db.priceHistory
    .where('ticker')
    .equals(ticker)
    .and(entry => entry.date >= startDate.toISOString().split('T')[0])
    .toArray();
}

export async function savePriceHistory(ticker, historyData) {
  const entries = historyData.map(h => ({
    ticker,
    ...h,
  }));
  return db.priceHistory.bulkAdd(entries, { allKeys: true });
}

export async function saveReport(reportData) {
  return db.reports.add({
    ...reportData,
    createdAt: new Date().toISOString(),
  });
}

export async function getReportsByType(type) {
  return db.reports.where('type').equals(type).toArray();
}

// 캐시 유효성 확인 (1일 이상 지나면 갱신)
export async function isCacheValid(ticker, hours = 24) {
  const latest = await db.priceHistory
    .where('ticker')
    .equals(ticker)
    .last();

  if (!latest) return false;
  const diffHours = (new Date() - new Date(latest.date)) / (1000 * 60 * 60);
  return diffHours < hours;
}
```

### 2.3 Zustand Store에 IndexedDB 동기화

**경로**: `src/store/journalStore.js` (수정)

```javascript
import { create } from 'zustand';
import { persist } from 'zustand/middleware';
import immer from 'immer';
import { db, addTransaction, getTransactionsByTicker } from '@/utils/db';

export const useJournalStore = create(
  persist(
    immer((set, get) => ({
      entries: [],

      // 일지 추가
      addEntry: async (entry) => {
        set(state => {
          state.entries.push(entry);
        });
        // IndexedDB에 동기 저장
        await db.transactions.add(entry);
      },

      // IndexedDB에서 데이터 로드 (앱 시작 시)
      loadFromDB: async () => {
        const dbEntries = await db.transactions.toArray();
        set({ entries: dbEntries });
      },

      // 종목별 일지 조회 (IndexedDB 활용)
      getEntriesByTicker: async (ticker) => {
        return await getTransactionsByTicker(ticker);
      },

      // ... 기존 메서드들
    })),
    {
      name: 'journal-store',
      version: 1,
    }
  )
);

// 앱 초기화 시 호출
export async function initializeJournalStore() {
  await useJournalStore.getState().loadFromDB();
}
```

```javascript
export default function InsightsCard({ entries, dateRange }) {
```

```javascript
import { useState } from 'react';
import { Card } from '@/components/ui/card';

import { Dialog, DialogContent, DialogHeader, DialogTitle, DialogTrigger } from '@/components/ui/dialog';

import { useWatchlistStore } from '@/store/watchlistStore';
import { useQuery } from '@tanstack/react-query';

export default function Watchlist() {
  const watchlist = useWatchlistStore(s => s.watchlist);
  const addToWatchlist = useWatchlistStore(s => s.addToWatchlist);
  const removeFromWatchlist = useWatchlistStore(s => s.removeFromWatchlist);
  const [newTicker, setNewTicker] = useState('');
  const [alertDialogOpen, setAlertDialogOpen] = useState(false);
  const [selectedStock, setSelectedStock] = useState(null);

    queryKey: ['watchlist-prices', watchlist],

            <DialogHeader>
              <DialogTitle>종목 추가</DialogTitle>
            </DialogHeader>
            <div className="flex gap-2">
              <Input
                placeholder="티커 입력"
                value={newTicker}
                onChange={e => setNewTicker(e.target.value)}
              />
              <Button onClick={handleAddStock}>추가</Button>

        <Table>

            <TableRow>
              <TableHead>종목명</TableHead>
              <TableHead>현재가</TableHead>
              <TableHead>변동률</TableHead>
              <TableHead>거래량</TableHead>
              <TableHead>액션</TableHead>

function MarketBriefing() {
  const [briefing, setBriefing] = useState(null);
```

### 4.2 Alert 스토어 확장

**경로**: `src/store/watchlistStore.js` (수정)

```javascript
export const useWatchlistStore = create(
  persist(
    immer((set) => ({
      watchlist: [],
      alerts: [], // 신규: 알림 조건

      addToWatchlist: (item) => set(state => {
        state.watchlist.push(item);
      }),

      removeFromWatchlist: (ticker) => set(state => {
        state.watchlist = state.watchlist.filter(item => item.ticker !== ticker);
      }),

      addAlert: (alert) => set(state => {
        // alert = { ticker, condition: 'price', value: 72000, type: 'above' }
        state.alerts.push(alert);
      }),

      removeAlert: (alertId) => set(state => {
        state.alerts = state.alerts.filter(a => a.id !== alertId);
      }),

      checkAlerts: (currentPrices) => {
        // 알림 조건 체크 로직
        // 조건 충족 시 browser notification API 실행
        return get().alerts.filter(alert => {
          const price = currentPrices[alert.ticker];
          if (!price) return false;

          if (alert.type === 'above') {
            return price >= alert.value;
          } else if (alert.type === 'below') {
            return price <= alert.value;
          }
          return false;
        });
      },
    })),
    { name: 'watchlist-store' }
  )
);
```
